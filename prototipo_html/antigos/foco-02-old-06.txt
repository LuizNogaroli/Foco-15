    <script>
        // Lógica para alternar entre as abas internas do imóvel
        document.addEventListener('click', function(e) {
            if (e.target.classList.contains('tab-link')) {
                const tabsContainer = e.target.closest('.imovel-content');
                
                // Resetar botões
                tabsContainer.querySelectorAll('.tab-link').forEach(btn => {
                    btn.classList.remove('active');
                    btn.style.background = '#e9ecef';
                    btn.style.color = '#333';
                });
                // Ocultar todos os painéis de aba
                tabsContainer.querySelectorAll('.tab-pane').forEach(pane => pane.style.display = 'none');
                
                // Ativar o botão clicado
                e.target.classList.add('active');
                e.target.style.background = '#0056b3';
                e.target.style.color = 'white';
                const tabId = e.target.getAttribute('data-tab');
                tabsContainer.querySelector('#' + tabId).style.display = 'block';
            }
        });
    </script>
