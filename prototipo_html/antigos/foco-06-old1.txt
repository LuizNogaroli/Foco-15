<script>
document.addEventListener("DOMContentLoaded", function () {
  const btnEnviar = document.getElementById("btnEnviarSPU");
  const modalEnvio = document.getElementById("modalEnvio");
  const btnConfirmarEnvio = document.getElementById("btnConfirmarEnvio");
  const btnCancelarEnvio = document.getElementById("btnCancelarEnvio");
  const form = document.getElementById("form06");

  if (btnEnviar && modalEnvio) {
    btnEnviar.addEventListener("click", function () {
      modalEnvio.style.display = "flex";
    });
  }

  if (btnCancelarEnvio && modalEnvio) {
    btnCancelarEnvio.addEventListener("click", function () {
      modalEnvio.style.display = "none";
    });
  }

  if (modalEnvio) {
    modalEnvio.addEventListener("click", function (event) {
      if (event.target === modalEnvio) {
        modalEnvio.style.display = "none";
      }
    });
  }

  if (btnConfirmarEnvio && form && modalEnvio) {
    btnConfirmarEnvio.addEventListener("click", function () {
      modalEnvio.style.display = "none";

      const eventoEnvio = new Event("submit", {
        bubbles: true,
        cancelable: true
      });

      form.dispatchEvent(eventoEnvio);
    });
  }
});
</script>
