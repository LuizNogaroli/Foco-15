    <script>
        document.addEventListener('DOMContentLoaded', function () {

            // ── 1. Hint popup ───────────────────────────────────────────────
            document.querySelectorAll('.hint-popup').forEach(function (icon) {
                function openPopup() {
                    var url = icon.getAttribute('data-popup');
                    if (!url) return;
                    window.open(
                        url, 'hintPopup',
                        'width=900,height=720,left=' + (screen.width / 2 - 450) +
                        ',top=' + (screen.height / 2 - 360)
                    );
                }
                icon.addEventListener('click', openPopup);
                icon.addEventListener('keydown', function (e) {
                    if (e.key === 'Enter' || e.key === ' ') {
                        e.preventDefault();
                        openPopup();
                    }
                });
            });

            // ── 2. Campos e regras de validação ─────────────────────────────
            const camposTexto = [
                { id: 'campo31',  erro: 'err31',  check: v => v !== '' },
                { id: 'campo32',  erro: 'err32',  check: v => v.trim().length >= 3 },
                { id: 'campo33',  erro: 'err33',  check: v => v !== '' },
                { id: 'campo34',  erro: 'err34',  check: v => v !== '' && parseInt(v) >= 0 },
                { id: 'campo35',  erro: 'err35',  check: v => v.trim().length >= 3 },
                { id: 'campo37',  erro: 'err37',  check: v => v !== '' && parseFloat(v) >= 0 },
                { id: 'campo38',  erro: 'err38',  check: v => v.trim().length >= 3 },
                { id: 'campo310', erro: 'err310', check: v => v.trim().length >= 2 },
                { id: 'campo311', erro: 'err311', check: v => v.trim().length >= 3 },
                { id: 'campo313', erro: 'err313', check: v => v.trim().length >= 3 },
            ];

            const gruposRadio = [
                { name: 'tipo_contrato', groupId: 'group-contrato', erro: 'err39' },
            ];

            // ── 3. Validação no submit ──────────────────────────────────────
            const form = document.getElementById('form03');

            form.addEventListener('submit', function (e) {
                e.preventDefault();
                let valido = true;

                // Valida campos de texto/select/number/textarea
                camposTexto.forEach(function (reg) {
                    const campo = document.getElementById(reg.id);
                    const erro = document.getElementById(reg.erro);
                    const ok = reg.check(campo.value);
                    erro.style.display = ok ? 'none' : 'block';
                    campo.classList.toggle('field-error', !ok);
                    campo.classList.toggle('field-valid', ok);
                    if (!ok) valido = false;
                });

                // Valida radio groups
                gruposRadio.forEach(function (grupo) {
                    const selecionado = document.querySelector(`input[name="${grupo.name}"]:checked`);
                    const groupEl = document.getElementById(grupo.groupId);
                    const erro = document.getElementById(grupo.erro);
                    const ok = !!selecionado;
                    erro.style.display = ok ? 'none' : 'block';
                    groupEl.classList.toggle('group-error', !ok);
                    if (!ok) valido = false;
                });

                if (valido) {
                    alert('✅ Seção 3 enviada com sucesso!');
                    // Substituir pelo envio real (fetch/AJAX)
                }
            });

            // ── 4. Botão Limpar ─────────────────────────────────────────────
            document.getElementById('btnLimpar').addEventListener('click', function () {
                if (confirm('Deseja limpar todos os campos?')) {
                    form.reset();
                    document.querySelectorAll('.error-msg').forEach(e => e.style.display = 'none');
                    document.querySelectorAll('.field-error, .field-valid').forEach(el => {
                        el.classList.remove('field-error', 'field-valid');
                    });
                    document.querySelectorAll('.group-error').forEach(el => {
                        el.classList.remove('group-error');
                    });
                }
            });

        });
    </script>
